邻接矩阵实现无向图的遍历

//邻接矩阵实现无向图的遍历
#include<bits/stdc++.h>
using namespace std;

const int MaxSize=10;
int visited[MaxSize]= {0};
template
class MGraph {
public:
MGraph(DataType a[],int n,int e); //构造函数
// ~MGraph(); //析构函数
void DFS(int v); //深度优先遍历图
void BFS(int v); //广度优先遍历图
private:
DataType vertex[MaxSize]; //存放图中顶点的数组
int edge[MaxSize][MaxSize]; //存放图中边的数组
int vertexNum,edgeNum; //图的顶点数和边数

};

template
MGraph::MGraph(DataType a[],int n,int e) {
int i,j,k;
vertexNum=n;
edgeNum=e;
for(i=0; i<vertexNum; i++) //储存顶点
vertex[i]=a[i];

}

template
void MGraph::DFS(int v) { //从顶点v出发
cout<<vertex[v]; //输出顶点
visited[v]=1; //标记已被访问过
for(int j=0; j<vertexNum; j++) {
if(edge[v][j]==1&&visited[j]==0) DFS(j);
}

}

template
void MGraph::BFS(int v) { //从顶点v出发
int w,j,Q[MaxSize]; //顺序队列 w:队头元素
int front=-1,rear=-1; //初始化队列
cout<<vertex[v];
visited[v]=1;
Q[++rear]=v; //被访问顶点入队
while(front!=rear) { //队列非空时
w=Q[++front];
for(j=0; j<vertexNum; j++) {
if(edge[w][j]==1&&visited[j]==0) {
cout<<vertex[j];
visited[j]=1;
Q[++rear]=j;
}
}
}

}

int main() {
int i;
char ch[]= {‘A’,’B’,’C’,’D’,’E’,’F’};
MGraphMG {ch,6,6}; //建立6个顶点，6条边的无向图,也可cin>>n,cin>>e输入顶点数和条数
for(i=0; i<MaxSize; i++) {
visited[i]=0;
}
cout<<”深度优先遍历序列是：”;
MG.DFS(0);
cout<<endl;
for(i=0; i<MaxSize; i++) {
visited[i]=0;
}
cout<<”广度优先遍历序列是：”;
MG.BFS(0);
return 0;
}
